#+begin_src sh :shebang #!/bin/sh :results output :mkdirp yes :tangle ./.git.sh

  cd "$(dirname -- "${0}")"

  CLEAN '.git.sh'

  GITADD 'README.org'

  git status ; git commit -m 'Started work on postgres and sea orm'
