#+begin_src sh :shebang #!/bin/sh :results output :tangle ./copy.sh
  cd "$('dirname' '--' "${0}")"
  echo 'START COPYING' \
      && find './' '-type' 'f' \
          | grep '^\./M_.*$' \
          | sed 's@^@("cp" "-vf" "--" "@g ; s@$@" "/usr/local/bin/");@g' \
          | sh \
  && echo 'DONE COPYING' ;
#+end_src
